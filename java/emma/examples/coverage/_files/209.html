464         * Returns a service section rapresentation in XMLParameter format

466         * @param stationKey search's key for current station
467         * @return service section rapresentation in XMLParameter format
468         * Author: Bertoli Marco
469         */
470        protected static XMLParameter getServiceSection(CommonModel model, Object stationKey) {
471                Vector classes = model.getClassKeys();
472                //creating set of service time distributions
473                XMLParameter[] distrParams = new XMLParameter[classes.size()];
474                Object currentClass;
475                for (int i = 0; i < distrParams.length; i++) {
476                        currentClass = classes.get(i);
477                        Object serviceDistribution = model.getServiceTimeDistribution(stationKey, currentClass);
478                        if (serviceDistribution instanceof Distribution) {
479                                // Service time is a Distribution and not a LDService
480                                distrParams[i] = DistributionWriter.getDistributionParameter((Distribution) serviceDistribution, model, currentClass);
481                        } else if (serviceDistribution instanceof ZeroStrategy) {
482                                // Zero Service Time Strategy --- Bertoli Marco
483                                distrParams[i] = new XMLParameter("ZeroTimeServiceStrategy", ZeroStrategy.getEngineClassPath(), model.getClassName(currentClass),
484                                                (String) null, true);
485                        } else {
486                                // Load dependent Service Time Strategy --- Bertoli Marco
487                                LDStrategy strategy = (LDStrategy) serviceDistribution;
488                                XMLParameter[] ranges = new XMLParameter[strategy.getRangeNumber()];
489                                Object[] rangeKeys = strategy.getAllRanges();
490 
491                                for (int j = 0; j < ranges.length; j++) {
492                                        // Creates "from" parameter
493                                        XMLParameter from = new XMLParameter("from", Integer.class.getName(), null,
494                                                        Integer.toString(strategy.getRangeFrom(rangeKeys[j])), true);
495                                        // Creates "distribution" parameter
496                                        XMLParameter[] distribution = DistributionWriter.getDistributionParameter(strategy.getRangeDistribution(rangeKeys[j]));
497                                        // Creates "function" parameter (mean value of the distribution)
498                                        XMLParameter function = new XMLParameter("function", String.class.getName(), null, strategy
499                                                        .getRangeDistributionMean(rangeKeys[j]), true);
500                                        ranges[j] = new XMLParameter("LDParameter", strategiesClasspathBase + serviceStrategiesSuffix + "LDParameter", null,
501                                                        new XMLParameter[] { from, distribution[0], distribution[1], function }, true);
502                                        // Sets array = false as it's not an array of equal elements
503                                        ranges[j].parameterArray = "false";
504                                }
505                                // Creates LDParameter array
506                                XMLParameter LDParameter = new XMLParameter("LDParameter", strategiesClasspathBase + serviceStrategiesSuffix + "LDParameter", null,
507                                                ranges, true);
508                                // Creates Service strategy
509                                distrParams[i] = new XMLParameter("LoadDependentStrategy", strategiesClasspathBase + serviceStrategiesSuffix
510                                                + "LoadDependentStrategy", model.getClassName(currentClass), new XMLParameter[] { LDParameter }, true);
511                                distrParams[i].parameterArray = "false";
512                        }
513                }
514                XMLParameter globalDistr = new XMLParameter("ServiceStrategy", strategiesClasspathBase + "ServiceStrategy", null, distrParams, false);
515                return globalDistr;
