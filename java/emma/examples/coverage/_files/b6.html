650         * jSIM architecture does not allow to correctly compute residence time for
651         * all visits.
652         * for this reason a scaling operation is needed.

657 
658        private void scaleMeasureWithVisitRatio() {
659                //retrieves JobClass object using its name
660                JobClass jobClass = network.getJobClass(jobClassName);
661                double visitRatio = 1.0;
662                //current measure node
663                NetNode thisNode = network.getNode(nodeName);
664 
665                // If measure is class specific
666                if (jobClass != null) {
667                        //reference node for this job class
668                        String referenceNodeName = jobClass.getReferenceNodeName();
669                        NetNode referenceNode = network.getNode(referenceNodeName);
670                        int visitsReferenceNode, visitsThisNode;
671                        try {
672                                //visits to the reference node
673                                NodeSection refOutputSection = referenceNode.getSection(NodeSection.OUTPUT);
674                                visitsReferenceNode = refOutputSection.getIntSectionProperty(NodeSection.PROPERTY_ID_ARRIVED_JOBS, jobClass);
675 
676                                //visits to this node node
677                                NodeSection thisNodeInputSection = thisNode.getSection(NodeSection.INPUT);
678 
679                                visitsThisNode = thisNodeInputSection.getIntSectionProperty(NodeSection.PROPERTY_ID_ARRIVED_JOBS, jobClass);
680 
681                                // the reference source generates a certain number of jobs
682                                // - some of them have reached this station
683                                // - other have been dropped before enetering this station or
684                                // elsewhere in the network
685                                int droppedJobs = network.getDroppedJobs(jobClass);
686 
687                                if (thisNodeInputSection instanceof BlockingQueue) {
688 
689                                        // if the input section is a BlockingQueue,
690                                        // remove (from the total number of visits)
691                                        // the jobs dropped by the BlockingQueue itself,
692                                        // because they have been counted
693                                        // in arrived jobs
694                                        visitsThisNode -= ((BlockingQueue) thisNode.getSection(NodeSection.INPUT)).getDroppedJobPerClass(jobClass.getId());
695 
696                                }
697 
698                                //visit ratio
699                                //
700                                // we must consider only not dropped jobs:
701                                // --> visit ratio = visits to this node / (visits to ref node - dropped jobs)
702                                visitRatio = (double) (visitsThisNode) / (visitsReferenceNode - droppedJobs);
703 
704                                //System.out.println("refNode: " + Integer.toString(visitsReferenceNode));
705                                //System.out.println("thisNode: " + Integer.toString(visitsThisNode));
706                                //System.out.println("dropped: " + droppedJobs);
707                                //System.out.println("visit ratio: " + Double.toString(visitRatio));
708 
709                        } catch (NetException ne) {
710                                visitRatio = 1;
711                                logger.error("Error in computing visit ratio.");
712                                ne.printStackTrace();
713                        }
714                }
715                // Measure is class-independent --> Bertoli Marco
716                else {
717                        try {
718                                Iterator it = network.getJobClasses().listIterator();
719                                double ratioSum = 0.0, weight = 0.0;
720                                while (it.hasNext()) {
721                                        JobClass jobC = (JobClass) it.next();
722                                        //reference node for this job class
723                                        String referenceNodeName = jobC.getReferenceNodeName();
724                                        NetNode referenceNode = network.getNode(referenceNodeName);
725                                        int visitsReferenceNode, visitsThisNode;
726                                        //visits to the reference node
727                                        NodeSection refOutputSection = referenceNode.getSection(NodeSection.OUTPUT);
728                                        visitsReferenceNode = refOutputSection.getIntSectionProperty(NodeSection.PROPERTY_ID_ARRIVED_JOBS, jobC);
729 
730                                        //visits to this node node
731                                        NodeSection thisNodeInputSection = thisNode.getSection(NodeSection.INPUT);
732 
733                                        visitsThisNode = thisNodeInputSection.getIntSectionProperty(NodeSection.PROPERTY_ID_ARRIVED_JOBS, jobC);
734 
735                                        int droppedJobs = network.getDroppedJobs(jobC);
736 
737                                        if (thisNodeInputSection instanceof BlockingQueue) {
738                                                visitsThisNode -= ((BlockingQueue) thisNode.getSection(NodeSection.INPUT)).getDroppedJobPerClass(jobC.getId());
739 
740                                        }
741                                        double currVisitRatio = (double) (visitsThisNode) / (visitsReferenceNode - droppedJobs);
742 
743                                        // Uses new calculated visit ratio to compute a weighted visit ratio
744                                        // This is done only if currVisitRatio value is valid
745                                        if (!Double.isNaN(currVisitRatio) && !Double.isInfinite(currVisitRatio)) {
746                                                ratioSum += currVisitRatio * visitsThisNode;
747                                                weight += visitsThisNode;
748                                                visitRatio = ratioSum / weight;
749                                        }
750                                }
751                        } catch (NetException ne) {
752                                visitRatio = 1;
753                                logger.error("Error in computing visit ratio.");
754                                logger.error(ne);
755                        }
756                }
757 
758                //correggere mean upper lower ecc....
759 
760                //sets scaled = true (to avoid other scaling operations)
761                scaled = true;
762                scaleFactor = visitRatio;
763 
764        }
