493                        // if instance is not correct returns false
494                        if (valueClass.isInstance(value)) {
495                                // If checker exists and its check is not satisfied returns false

499                                        }
500                                }

504                        }

508                /**
509                 * Sets value for this parameter. If parameter class is String sets it, otherwise
510                 * try to parse string if parameter is of a known numeric value. Returns if parameter
511                 * could be assigned correctly.
512                 * @param value to be assigned to this parameter or to be parsed
513                 * @return If value.class is different from valueClass, or parameter value is invalid,
514                 * or is not parsable, returns false, otherwise true
515                 */
516                public boolean setValue(String value) {
517                        // If parameter is of the correct type, sets it and return
518                        if (setValue((Object) value)) {
519                                return true;
520                        }
521 
522                        // Otherwise if it's a known number, try to decode it
523                        Object objval = null;
524                        try {
525                                if (valueClass.equals(Integer.class)) {
526                                        objval = Integer.decode(value);
527                                } else if (valueClass.equals(Long.class)) {
528                                        objval = Long.decode(value);
529                                } else if (valueClass.equals(Short.class)) {
530                                        objval = Short.decode(value);
531                                } else if (valueClass.equals(Byte.class)) {
532                                        objval = Byte.decode(value);
533                                } else if (valueClass.equals(Float.class)) {
534                                        objval = new Float(Float.parseFloat(value));
535                                } else if (valueClass.equals(Double.class)) {
536                                        objval = new Double(Double.parseDouble(value));
537                                } else if (valueClass.equals(Boolean.class)) {
538                                        objval = new Boolean(value);
539                                } else {
540                                        return false;
541                                }
542                                return setValue(objval);
543                        } catch (NumberFormatException e) {
544                                return false;
545                        }
546                }
