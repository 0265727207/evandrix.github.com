586        /**
587         * This function is called each time from, to or iterations field is changed.
588         */
589        private void updateFields() {
590                String type = (String) this.type.getSelectedItem();
591                double from = current;
592                // From field is 'current' value if parsing fails.
593                try {
594                        from = Double.parseDouble(this.from.getText());
595                        if (from < 0) {
596                                from = 0.0;
597                        }
598                } catch (NumberFormatException ex) {
599                        // Do nothing
600                }
601                try {
602                        double to = Double.parseDouble(this.to.getText());
603                        if (to < 0) {
604                                to = 0.0;
605                        }
606                        int iterations = 2;
607                        if (this.iterations.getText() != null && !this.iterations.getText().equals("")) {
608                                iterations = Integer.parseInt(this.iterations.getText());
609                        }
610                        if (iterations < 2) {
611                                iterations = 2;
612                        }
613                        int classNum = classNames.get(className.getSelectedItem()).intValue();
614                        int stationNum = -1;
615                        // If from and to are expressed as percentage, divides them by 100
616                        if (classNum < 0) {
617                                from /= 100;
618                                to /= 100;
619                        }
620                        // Correct values for population mix what-if analysis
621                        if (type.equals(ExactConstants.WHAT_IF_MIX)) {
622                                if (from > 1) {
623                                        from = 0;
624                                }
625                                if (to > 1) {
626                                        to = 1;
627                                }
628                        }
629 
630                        if (stationName.getSelectedItem() != null) {
631                                stationNum = stationNames.get(stationName.getSelectedItem()).intValue();
632                        }
633                        values = data.generateWhatIfValues(type, from, to, iterations, classNum, stationNum);
634                } catch (NumberFormatException e) {
635                        // Number format is wrong, skips this update.
636                } catch (ArithmeticException ex) {
637                        JOptionPane.showMessageDialog(this, "Closed class with 0 customers found. They will be set to 1.", "Error", JOptionPane.ERROR_MESSAGE);
638                        double[] classData = data.getClassData().clone();
639                        for (int i = 0; i < classData.length; i++) {
640                                if (data.getClassTypes()[i] == ExactConstants.CLASS_CLOSED && classData[i] < 1) {
641                                        classData[i] = 1.0;
642                                }
643                        }
644                        data.setClassData(classData);
645                        updateFields();
646                }
647                setFromToIterations();
648        }
