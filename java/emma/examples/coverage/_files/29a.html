552        private void startSimulation() {
553                //if simulation is not in pause state
554                if (!SIM_STOP.isEnabled()) {
555                        // Asks for confirmation before overwriting previous simulation data
556                        if (model.containsSimulationResults()) {
557                                // Find frame to show confirm dialog
558                                Component parent = this;
559                                if (resultsWindow != null && resultsWindow.isFocused()) {
560                                        parent = resultsWindow;
561                                }
562 
563                                int resultValue = JOptionPane.showConfirmDialog(parent, "This operation will overwrite old simulation results." + "Continue anyway?",
564                                                "JMT - Warning", JOptionPane.YES_NO_OPTION, JOptionPane.WARNING_MESSAGE);
565                                if (resultValue == JOptionPane.NO_OPTION) {
566                                        return;
567                                }
568                        }
569 
570                        // Correct eventual problems on preloading for closed classes
571                        model.manageJobs();
572                        // Removes previous ResultsWindow
573                        if (resultsWindow != null) {
574                                resultsWindow.dispose();
575                                SHOW_RESULTS.setEnabled(false);
576                        }
577                        mc = new ModelChecker(model, model, model, model, false);
578                        pw.setModelChecker(mc);
579                        //pw.setToJMVAConversion(false);
580                        pw.updateProblemsShown(false);
581                        if (!mc.isEverythingOkNormal()) {
582                                WindowListener wl = pw.getWindowListeners()[0];
583                                pw.show();
584                        } else {
585                                pw.setVisible(false);
586                                launchSimulation();
587                        }
588 
589                } else {
590                        if (!model.isParametricAnalysisEnabled()) {
591                                dispatcher.restartSimulation();
592                        } else {
593                                batchThread.restartSimulation();
594                        }
595                }
596        }

598        public void launchSimulation() {
599                if (!model.isParametricAnalysisEnabled()) {
600                        try {
601                                File temp = File.createTempFile("~JModelSimulation", ".xml");
602                                temp.deleteOnExit();
603                                XMLWriter.writeXML(temp, model);

606                                SHOW_RESULTS.setEnabled(true);
607                                dispatcher = new DispatcherThread(this, model, (ResultsModel) model.getSimulationResults());
608                                dispatcher.startSimulation(temp);
609                        } catch (Exception e) {
610                                handleException(e);
611                        }
612                } else {
613                        SHOW_RESULTS.setEnabled(false);
614                        // Removes previous ResultsWindow
615                        //if (parametricResultsWindow != null) {
616                        //    parametricResultsWindow.dispose();
617                        //}

620                        }
