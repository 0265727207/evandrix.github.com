456        static public double incompleteGammaComplement(double a, double x) throws ArithmeticException {

462                }

465                        return 1.0 - incompleteGamma(a, x);
466                }

470                        return 0.0;
471                }

474

476                y = 1.0 - a;
477                z = x + y + 1.0;
478                c = 0.0;
479                pkm2 = 1.0;
480                qkm2 = x;
481                pkm1 = x + 1.0;
482                qkm1 = z * x;
483                ans = pkm1 / qkm1;
484 
485                do {
486                        c += 1.0;
487                        y += 1.0;
488                        z += 2.0;
489                        yc = y * c;
490                        pk = pkm1 * z - pkm2 * yc;
491                        qk = qkm1 * z - qkm2 * yc;
492                        if (qk != 0) {
493                                r = pk / qk;
494                                t = Math.abs((ans - r) / r);
495                                ans = r;
496                        } else {
497                                t = 1.0;
498                        }
499 
500                        pkm2 = pkm1;
501                        pkm1 = pk;
502                        qkm2 = qkm1;
503                        qkm1 = qk;
504                        if (Math.abs(pk) > big) {

506                                pkm1 *= biginv;
507                                qkm2 *= biginv;
508                                qkm1 *= biginv;
509                        }
510                } while (t > MACHEP);
511 
512                return ans * ax;
513        }
