198        /**
199         * Creates the choosen parameter option panel
200         * @param pad the instance of <code>ParameterAnalysisDefinition</code> model.
201         * @return  the <code>ParameterOptionPanel</code> corresponding to the <code>ParameterAnalysisDefinition</code>
202         * passed as parameter
203         */
204        protected ParameterOptionPanel createPanel(ParametricAnalysisDefinition pad) {
205                ParameterOptionPanel pop;
206                if (pad == null) {
207                        pop = new EmptyPanel();
208                } else {
209                        if (pad instanceof NumberOfCustomerParametricAnalysis) {
210                                pop = new NumberOfCustomersPanel((NumberOfCustomerParametricAnalysis) pad, cd, sd, simd, gui);
211                        } else if (pad instanceof PopulationMixParametricAnalysis) {
212                                pop = new PopulationMixPanel((PopulationMixParametricAnalysis) pad, cd, sd, simd);
213                        } else if (pad instanceof ServiceTimesParametricAnalysis) {
214                                pop = new ServiceTimesPanel((ServiceTimesParametricAnalysis) pad, cd, sd, simd);
215                        } else if (pad instanceof ArrivalRateParametricAnalysis) {
216                                pop = new ArrivalRatesPanel((ArrivalRateParametricAnalysis) pad, cd, sd, simd);
217                        } else if (pad instanceof SeedParametricAnalysis) {
218                                pop = new SeedPanel((SeedParametricAnalysis) pad, cd, sd, simd);
219                        } else {
220                                pop = null;
221                        }
222                }
223                return pop;
224        }
