328        /**
329         * Check if the face is visible or not. Due to the convex solid property, a
330         * positive dot product means that the face has the normal in the same
331         * direction of the sight, thus it is not visible because there exist faces
332         * which cover the sight of the analyzed one.
333         * 
334         * @param faces
335         *            Faces to split.
336         * @param forward
337         *            Viewing direction.
338         */
339        protected void splitByVisibility(Vector4 forward) {
340                forward = forward.unit();
341                visibleFacesSet.clear();
342                hiddenFacesSet.clear();
343                visibleEdgesSet.clear();
344                hiddenEdgesSet.clear();
345                visibleVerticesSet.clear();
346                hiddenVerticesSet.clear();
347                hullEdgesSet.clear();
348                hullVerticesSet.clear();
349                maxVertexDistance = 1.0f;
350                centroid.set(Vector4.createPoint3D(0.0f, 0.0f, 0.0f));
351                float[] max = { 0.0f, 0.0f, 0.0f };
352                float[] min = { 0.0f, 0.0f, 0.0f };
353 
354                int size = faces.size();
355                for (int i = 0; i < size; i++) {
356                        HullFace face = faces.get(i);
357 
358                        Vector4 normal = face.getNormal();
359                        HullEdge[] edges = face.getEdges();
360                        HullVertex[] vertices = face.getVertices();
361 
362                        float facing = normal.dot(forward);
363                        boolean isVisible = (facing <= VISIBILITY_EPSILON);
364                        if (isVisible) {
365                                visibleFacesSet.add(face);
366                        } else {
367                                hiddenFacesSet.add(face);
368                        }
369                        face.setShade(facing);
370 
371                        // Create unique and visibility sets
372                        for (int j = 0; j < edges.length; j++) {
373                                HullEdge e = edges[j];
374                                if (isVisible) {
375                                        visibleEdgesSet.add(e);
376 
377                                        // Fix edges flagged as hidden but actually visible
378                                        if (hiddenEdgesSet.contains(e)) {
379                                                hiddenEdgesSet.remove(e);
380                                        }
381                                } else if (!visibleEdgesSet.contains(e)) {
382                                        // Add only if no other faces have already claimed the edge
383                                        // visibility.
384                                        hiddenEdgesSet.add(e);
385                                }
386                                if (hullEdgesSet.add(e)) {
387                                }
388                        }
389                        for (int j = 0; j < vertices.length; j++) {
390                                HullVertex v = vertices[j];
391                                Vector4 worldPos = Vector4.createPoint3D(v.x(), v.y(), v.z());
392                                Vector4 screenPos = prj.project(worldPos);
393                                PointData p = new PointData(worldPos, screenPos, v);
394                                if (isVisible) {
395                                        visibleVerticesSet.add(p);
396 
397                                        // Fix vertices flagged as hidden but actually visible
398                                        if (hiddenVerticesSet.contains(p)) {
399                                                hiddenVerticesSet.remove(p);
400                                        }
401                                } else if (!visibleVerticesSet.contains(p)) {
402                                        // Add only if no other faces have already claimed the
403                                        // vertex visibility.
404                                        hiddenVerticesSet.add(p);
405                                }
406                                if (p.vertex.getID() >= 0 && hullVerticesSet.add(p)) {
407                                        if (v.x() > max[0]) {
408                                                max[0] = v.x();
409                                        }
410                                        if (v.y() > max[1]) {
411                                                max[1] = v.y();
412                                        }
413                                        if (v.z() > max[2]) {
414                                                max[2] = v.z();
415                                        }
416                                        if (v.x() < min[0]) {
417                                                min[0] = v.x();
418                                        }
419                                        if (v.y() < min[1]) {
420                                                min[1] = v.y();
421                                        }
422                                        if (v.z() < min[2]) {
423                                                min[2] = v.z();
424                                        }
425                                }
426                        }
427                }
428 
429                // Centering and scaling
430                centroid.x((max[0] + min[0]) * 0.5f * zoomFactor);
431                centroid.y((max[1] + min[1]) * 0.5f * zoomFactor);
432                centroid.z((max[2] + min[2]) * 0.5f * zoomFactor);
433                max[0] -= min[0];
434                max[1] -= min[1];
435                max[2] -= min[2];
436                maxVertexDistance = 0.5f * (float) Math.sqrt(max[0] * max[0] + max[1]
437                                * max[1] + max[2] * max[2]);
438                centroid.divSelf(maxVertexDistance * zoomFactor);
439 
440                // Build the verticesInside set.
441                verticesInsideSet.clear();
442                for (int i = 0; i < allVerticesArray.length; i++) {
443                        HullVertex v = allVerticesArray[i];
444                        PointData p = new PointData(null, null, v); // Dummy PointData, just
445                                                                                                                // to check the HashSet
446                        if (!hullVerticesSet.contains(p)) {
447                                Vector4 worldPos = Vector4.createPoint3D(v.x(), v.y(), v.z());
448                                Vector4 screenPos = prj.project(worldPos);
449                                p = new PointData(worldPos, screenPos, v);
450                                verticesInsideSet.add(p);
451                        }
452                }
453 
454                // TODO: Remove useless vertices
455 
456                // Convert sets into arrays for easy modifies
457                hullEdgesArray = hullEdgesSet
458                                .toArray(new HullEdge[hullEdgesSet.size()]);
459                hullVerticesArray = hullVerticesSet
460                                .toArray(new PointData[hullVerticesSet.size()]);
461                hullFacesArray = this.faces.toArray(new HullFace[this.faces.size()]);
462 
463                visibleFacesArray = visibleFacesSet
464                                .toArray(new HullFace[visibleFacesSet.size()]);
465                hiddenFacesArray = hiddenFacesSet.toArray(new HullFace[hiddenFacesSet
466                                .size()]);
467                visibleEdgesArray = visibleEdgesSet
468                                .toArray(new HullEdge[visibleEdgesSet.size()]);
469                hiddenEdgesArray = hiddenEdgesSet.toArray(new HullEdge[hiddenEdgesSet
470                                .size()]);
471                visibleVerticesArray = visibleVerticesSet
472                                .toArray(new PointData[visibleVerticesSet.size()]);
473                hiddenVerticesArray = hiddenVerticesSet
474                                .toArray(new PointData[hiddenVerticesSet.size()]);
475                verticesInsideArray = verticesInsideSet
476                                .toArray(new PointData[verticesInsideSet.size()]);
477 
478                // Sort sets into separate arrays
479                sortedVisibleVertices = visibleVerticesArray.clone();
480                Arrays.sort(sortedVisibleVertices);
481                sortedHiddenVertices = hiddenVerticesArray.clone();
482                Arrays.sort(sortedHiddenVertices);
483                sortedVerticesInside = verticesInsideArray.clone();
484                Arrays.sort(sortedVisibleVertices);
485 
486                countFaceVertices();
487        }

489        protected void countFaceVertices() {
490                /*
491                 * Per ogni faccia faccia.contatore = 0; faccia.marcata = false;
492                 * 
493                 * Per ogni vertice Ä convex hull Per ogni faccia Se vertice Ä faccia
494                 * incrementaFaccia(faccia) Per ogni faccia faccia.marcata = false
495                 */
496 
497                // Reset vertex counters and marks
498                for (int fi = 0; fi < hullFacesArray.length; fi++) {
499                        HullFace f = hullFacesArray[fi];
500                        f.setNumCoplanarVertices(0);
501                }
502 
503                // Start propagating the number of vertices per face
504                for (int vi = 0; vi < hullVerticesArray.length; vi++) {
505                        // Cleanup marked bits
506                        for (int fi = 0; fi < hullFacesArray.length; fi++) {
507                                HullFace f = hullFacesArray[fi];
508                                f.setMarked(false);
509                        }
510 
511                        for (int fi = 0; fi < hullFacesArray.length; fi++) {
512                                HullVertex v = hullVerticesArray[vi].vertex;
513                                HullFace f = hullFacesArray[fi];
514 
515                                // Start color propagation
516                                if (v.getID() == f.getVertex(0).getID()
517                                                || v.getID() == f.getVertex(1).getID()
518                                                || v.getID() == f.getVertex(2).getID()) {
519                                        if (!f.isMarked()) {
520                                                incrementFaceCounter(f);
521                                        }
522                                }
523                        }
524                }

527        protected void incrementFaceCounter(HullFace face) {
528                /*
529                 * faccia.contatore++ faccia.marcata = true
530                 * 
531                 * Per ogni spigolo Ä faccia Per ogni faccia adiacente Ä spigolo Se
532                 * adiacente != questa && adiacente coplanare faccia &&
533                 * !adiacente.marcata incrementaFaccia(adiacente)
534                 */
535 
536                // Increment the counter and set as marked
537                face.setNumCoplanarVertices(1 + face.getNumCoplanarVertices());
538                face.setMarked(true);
539 
540                for (int ei = 0; ei < 3; ei++) {
541                        HullEdge e = face.getEdge(ei);
542                        for (int ai = 0; ai < 2; ai++) {
543                                HullFace adj = e.getFace(ai);
544                                /*
545                                 * if ( face.volumeSign( adj.getVertex( 0 ) ) == 0 &&
546                                 * face.volumeSign( adj.getVertex( 1 ) ) == 0 &&
547                                 * face.volumeSign( adj.getVertex( 2 ) ) == 0 &&
548                                 * !face.isMarked() ) { incrementFaceCounter( adj ); }
549                                 */
550                                if (!adj.isMarked() && face.isCoplanar(adj)) {
551                                        incrementFaceCounter(adj);
552                                }
553                        }
554                }
