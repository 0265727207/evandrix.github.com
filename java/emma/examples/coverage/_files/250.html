304        /**
305         * It acts like Graphics.fillArc(...), I have written this method
306         * since the predifined Java function presented strange behavior with
307         * very small angles
308         *
309         * @param g  the Graphics reference
310         * @param x  see Graphics.fillArc(...)
311         * @param y  see Graphics.fillArc(...)
312         * @param diameter  the diameter of the arc
313         * @param startAngle  the starting angle in degrees (0? is interpreted as 3 o'clock)
314         * @param angle  the angle width
315         */
316        private void myFillArc(Graphics g, int x, int y, int diameter, int startAngle, int angle) {
317                double c = 2 * Math.PI / 360;
318                int radius = diameter / 2;
319                double MIN_ANGLE = 1 * c;
320                double nPoints = Math.round(angle * c / (float) MIN_ANGLE);
321                double xStart = x + radius;
322                double yStart = y + radius;
323                double reachedAngle = startAngle * c;
324                for (int i = 0; i < nPoints; i++) {
325                        double x1;
326                        double y1;
327                        double x2;
328                        double y2;
329                        x1 = radius * Math.cos(reachedAngle);
330                        y1 = radius * Math.sin(reachedAngle);
331                        x2 = radius * Math.cos(reachedAngle + MIN_ANGLE);
332                        y2 = radius * Math.sin(reachedAngle + MIN_ANGLE);
333                        x1 = xStart + x1;
334                        y1 = yStart - y1;
335                        x2 = xStart + x2;
336                        y2 = yStart - y2;
337                        x1 = Math.round(x1);
338                        y1 = Math.round(y1);
339                        x2 = Math.round(x2);
340                        y2 = Math.round(y2);
341                        int[] xValues = { (int) xStart, (int) x1, (int) x2 };
342                        int[] yValues = { (int) yStart, (int) y1, (int) y2 };
343                        g.fillPolygon(xValues, yValues, 3);
344                        reachedAngle += MIN_ANGLE;
345                }
