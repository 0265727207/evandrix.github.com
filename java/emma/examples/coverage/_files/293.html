524         * Returns given name if a station with the same name does not exists or makes it unique
525         * @param name station name
526         * @return unique name
527         */
528        private String getUniqueStationName(String name) {
529                TreeSet<String> names = new TreeSet<String>(); // Map of all unique names with their first users
530                Vector keys = getStationKeys();
531                // Finds all used names
532                for (int i = 0; i < keys.size(); i++) {
533                        names.add(this.getStationName(keys.get(i)));
534                }
535 
536                // If name is new, returns it
537                if (!names.contains(name)) {
538                        return name;
539                }
540 
541                int num;
542                // If format is already '*_[number]' increment number
543                char[] charname = name.toCharArray();
544                int n = charname.length - 1;
545                while (charname[n] >= '0' && charname[n] <= '9' && n > 0) {
546                        n--;
547                }
548                if (charname[n] == '_') {
549                        num = Integer.parseInt(name.substring(n + 1));
550                        name = name.substring(0, n); // Removes suffix
551                }
552                // Otherwise uses number 1
553                else {
554                        num = 1;
555                }
556                // Finds unique number
557                while (names.contains(name + "_" + num)) {
558                        num++;
559                }
560                return name + "_" + num;
561        }
