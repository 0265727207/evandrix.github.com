431                // Sanity checks on input model.
432                if (model.getWhatIfClass() >= 0 && model.getClassTypes()[model.getWhatIfClass()] != ExactConstants.CLASS_OPEN) {
433                        throw new InputDataException("Cannot change arrival rate of a closed class.");
434                }

437                }

441

444 
445                // Iterates for what-if executions
446                for (int i = 0; i < model.getWhatIfValues().length && !stopped; i++) {
447                        double[] current = initials.clone();
448                        // If this is one class only
449                        if (model.getWhatIfClass() >= 0) {
450                                current[model.getWhatIfClass()] = values[i];
451                        }

456                                                current[j] = initials[j] * values[i];
457                                        }
458                                }
459                        }
460                        model.setClassData(current);
461

463                        if (stopped) {
464                                break;
465                        }
466 
467                        // Now solves current model - we cannot interrupt this as it's not designed to be done.
468                        finalDispatch(model, i);
469                }
470                // Resets initial model
471                model.setClassData(initials);
472                // Results are ok if the process was not stopped.
473                model.setResultsOK(!stopped);
474        }

578                        else {
579                                for (int j = 0; j < model.getClasses(); j++) {

581                                }
582                        }

601         * @param model input model
602         */
603        private void whatIfMix(ExactModel model) throws SolverException, InputDataException {
604                // First and second closed class for population mix what-if
605                int class1, class2 = -1;
606                class1 = model.getWhatIfClass();
607                if (class1 < 0) {
608                        throw new InputDataException("Class not specified for population mix what-if analysis.");
609                }
610                // Find second class
611                for (int i = 0; i < model.getClasses(); i++) {
612                        if (model.getClassTypes()[i] == ExactConstants.CLASS_CLOSED && i != class1) {
613                                if (class2 < 0) {
614                                        class2 = i;
615                                } else {
616                                        throw new InputDataException("Only models with two closed classes are supported. More than two classes detected.");
617                                }
618                        }
619                }
620                if (class2 < 0) {
621                        throw new InputDataException("Only models with two closed classes are supported. Only one classes detected.");
622                }
623 
624                // Values for what-if
625                double[] values = model.getWhatIfValues();
626 
627                // Backup class datas
628                double[] initials = model.getClassData().clone();
629 
630                // Value for total number of customer
631                double N = initials[class1] + initials[class2];
632 
633                // Iterates for what-if executions
634                int i;
635                for (i = 0; i < model.getWhatIfValues().length && !stopped; i++) {
636                        double[] current = initials.clone();
637                        current[class1] = values[i] * N;
638                        current[class2] = (1 - values[i]) * N;
639                        // Check for not integer values
640                        if (Math.abs(current[class1] - Math.rint(current[class1])) > 1e-8) {
641                                throw new InputDataException("A fractional population value was assigned to class " + model.getClassNames()[class1] + " during step "
642                                                + i);
643                        } else if (Math.abs(current[class2] - Math.rint(current[class2])) > 1e-8) {
644                                throw new InputDataException("A fractional population value was assigned to class " + model.getClassNames()[class2] + " during step "
645                                                + i);
646                        }
647                        // Rounds number to avoid truncation problems
648                        current[class1] = Math.round(current[class1]);
649                        current[class2] = Math.round(current[class2]);
650 
651                        model.setClassData(current);
652

654                        if (stopped) {
655                                break;
656                        }
657 
658                        // Now solves current model - we cannot interrupt this as it's not designed to be done.

660                }

664                model.setResultsOK(!stopped);
