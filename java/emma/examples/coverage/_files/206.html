536        /**
537         * Parses A Parameter Array node, returning a Map of ClassName -> subParameter
538         * @param parameterNode
539         * @return a Map of ClassName -> subParameter
540         */
541        protected static Map<String, Node> parseParameterRefclassArray(Element parameterNode) {
542                // For some reasons getElementsByTagName returns only first service time strategy.
543                // So we need to look every children of parameterNode node.
544                TreeMap<String, Node> res = new TreeMap<String, Node>();
545                Node child = parameterNode.getFirstChild();
546                String refClass;
547                // This manual parsing is a bit unclean but works well and it's really fast.
548                // I was forced to do in this way for the problem said before.
549                while (child != null) {
550                        while (child != null && (child.getNodeType() != Node.ELEMENT_NODE || !child.getNodeName().equals(XML_E_PARAMETER_REFCLASS))) {
551                                child = child.getNextSibling();
552                        }
553 
554                        if (child == null) {
555                                break;
556                        }
557                        refClass = findText(child);
558                        // Now finds first subParameter element
559                        while (child != null && (child.getNodeType() != Node.ELEMENT_NODE || !child.getNodeName().equals(XML_E_SUBPARAMETER))) {
560                                child = child.getNextSibling();
561                        }
562 
563                        if (child == null) {
564                                break;
565                        }
566 
567                        // Puts className and subParameter into destination Map
568                        res.put(refClass, child);
569                        child = child.getNextSibling();
570                }
571 
572                return res;
573        }

621                for (Object className2 : classNames) {

624                        for (Object routStratKey : routStratKeys) {
625                                if (className.equals(engineToGuiRouting.get(routStratKey).getClassPath())) {

627                                }

631                        RoutingStrategy rs = (RoutingStrategy) model.getRoutingStrategy(key, classes.get(className2));
632                        if (rs instanceof ProbabilityRouting) {
633                                // Creates a Vector of all empirical entris. Could not be done automaticly
634                                // for the above problem with array (see parseParameterRefclassArray)
635                                Vector<Node> entries = new Vector<Node>();
636                                // Finds EntryArray node

639                                        entryArray = entryArray.getNextSibling();
640                                }

645                                        while (child != null && (child.getNodeType() != Node.ELEMENT_NODE || !child.getNodeName().equals(XML_E_SUBPARAMETER))) {
646                                                child = child.getNextSibling();
647                                        }

649                                                entries.add(child);
650                                                child = child.getNextSibling();
651                                        }
652                                }

661                                        Map<String, Double> tmp = new TreeMap<String, Double>();
662                                        tmp.put(stationName, probability);
663                                        // Put into empiricalRouting a pair of station key and class key and map with
664                                        // station names instead of station key
665                                        empiricalRouting.put(new Object[] { key, classes.get(className2) }, tmp);
666                                }

801                        if (distribution.size() == 0) {
