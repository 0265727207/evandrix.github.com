445        private int getDelimiterIndex(String s) {
446                char c = s.charAt(0);
447                String searchString;
448 
449                if (c == ' ') {
450                        searchString = "Space";
451                } else if (s.startsWith("\\t") || c == '\t') {
452                        searchString = "Tab";
453                } else if (s.startsWith("\\") == false) {
454                        searchString = "" + c;
455                } else {
456                        return 0;
457                }
458 
459                // Case 1: string in drop-down box.
460                for (int i = 0; i < loggerOutputDelimiterDropdown.length; i++) {
461                        if (loggerOutputDelimiterDropdown[i].equals(searchString) == true) {
462                                return i;
463                        }
464                }
465 
466                // Case 2: add a new character to dropdown box (extensibility)
467                delimiterChooser.insertItemAt(searchString, loggerOutputDelimiterDropdown.length - 1);
468                return (loggerOutputDelimiterDropdown.length - 1);
469        }

478        private void setDelimiterAndReplacement(boolean isGlobal) {
479                Character delim;
480                Integer autoAppendPolicy;
481 
482                // Ignores mistaken change from loggerChooser (when out of focus)
483                if (preventSaveReplacement == true) {
484                        return;
485                }
486 
487                // Sets the delimiter parameter for local log
488                if (loggerOutputDelimiterDropdown[delimiterChooser.getSelectedIndex()].equalsIgnoreCase("Space")) {
489                        delim = new Character(' ');
490                } else if (loggerOutputDelimiterDropdown[delimiterChooser.getSelectedIndex()].equalsIgnoreCase("Tab")) {
491                        delim = new Character('\t');
492                } else {
493                        delim = new Character(loggerOutputDelimiterDropdown[delimiterChooser.getSelectedIndex()].charAt(0));
494                }
495 
496                // Set the append/replace parameter for local log
497                String tmp = loggerAutoAppendDropdown[autoAppendChooser.getSelectedIndex()];
498                if (tmp.equalsIgnoreCase(NAME_loggerAutoAppendDropdownReplace)) {
499                        autoAppendPolicy = new Integer(LoggerParameters.LOGGER_AR_REPLACE);
500                } else if (tmp.equalsIgnoreCase(NAME_loggerAutoAppendDropdownAppend)) {
501                        autoAppendPolicy = new Integer(LoggerParameters.LOGGER_AR_APPEND);
502                } else if (tmp.equalsIgnoreCase(NAME_loggerAutoAppendDropdownAsk)) {
503                        autoAppendPolicy = new Integer(LoggerParameters.LOGGER_AR_ASK);
504                } else {
505                        autoAppendPolicy = new Integer(LoggerParameters.LOGGER_AR_ASK);
506                }
507                tmp = null;
508 
509                /* MF (Bertoli suggests): global append should be the only mode for 0.7.4 
510                if (isGlobal == false)
511                {
512                
513                        // If not logging to global.csv, set the delimiter and append policy per logger
514                        loggerParameters.delimiter = delim;
515                        loggerParameters.intAutoAppend = autoAppendPolicy;
516                        */
517 
518                // Finally, set the globally effective 'delimiter' and 'append' mode for logs
519                stationData.setLoggingGlbParameter("delim", delim.toString());
520                stationData.setLoggingGlbParameter("decimalSeparator", loggerOutputDecimalSeparatorDropdown[decimalSeparatorChooser.getSelectedIndex()]);
521                stationData.setLoggingGlbParameter("autoAppend", autoAppendPolicy.toString());
522        }

524        private String updateFileLabelStatus() {
525                String tmpStatus = "";
526                String path = loggerUniversalLogPath;
527 
528                // Make pathname pretty: remove the leading period, it confuses java.io.File.exists() method.
529                if ((path.equals(".") == true) || (path.equals("." + File.separator) == true)) {
530                        path = "";
531                }
532                // Add a trailing slash
533                if ((path.length() > 0) && (path.endsWith(File.separator) == false) && (path.endsWith("/") == false)) {
534                        path = path + File.separator;
535                }
536 
537                try {
538                        logFile = new File(path + loggerParameters.name);
539 
540                        if (logFile.exists() == true) {
541                                if (logFile.length() == 0) {
542                                        tmpStatus = "  (empty file)";
543                                } else {
544                                        tmpStatus = "  (" + logFile.length() / 1024 + "KB file)";
545                                }
546                        } else {
547                                tmpStatus = "  (does not exist).";
548                        }
549 
550                        labelFileStatus.setText("File:  " + logFile.getAbsolutePath() + " " + tmpStatus);
551                } catch (Exception e) {
552                        labelFileStatus.setText("Caught exception: " + e.getCause() + " trying to access " + loggerParameters.name + " File.");
553                }
554 
555                LoggerSectionPanel.this.doLayout();
556                LoggerSectionPanel.this.repaint();
557 
558                return tmpStatus;
559        }
