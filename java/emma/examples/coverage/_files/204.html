507        /**
508         * Inner class used to create simple file filters with only extension check
509         */
510        protected static class JmtFileFilter extends FileFilter {
511                public static final String SEP = ";";
512                private String description;
513                private String[] extensions;
514 
515                /**
516                 * Creates a new filefilter with specified dot and comma separated list of 
517                 * extensions and specified description
518                 * @param extensionList comma separated list of extensions 
519                 * of this filter (for example ".jmt;.jmodel")
520                 * @param description description of this filter
521                 */
522                public JmtFileFilter(String extensionList, String description) {
523                        this.extensions = extensionList.split(SEP);
524                        this.description = createDescription(description, extensions);
525                }
526 
527                /**
528                 * Whether the given file is accepted by this filter.
529                 */
530                @Override
531                public boolean accept(File f) {
532                        String name = f.getName().toLowerCase();
533                        if (f.isDirectory()) {
534                                return true;
535                        }
536                        for (String extension : extensions) {
537                                if (name.endsWith(extension)) {
538                                        return true;
539                                }
540                        }
541                        return false;
542                }
543 
544                /**
545                 * Creates a suitable description for this filter
546                 * @param text text description of the filter
547                 * @param extensions extensions to be matched
548                 * @return created description
549                 */
550                private String createDescription(String text, String[] extensions) {
551                        StringBuffer sb = new StringBuffer();
552                        sb.append(text);
553                        sb.append(" (*");
554                        for (int i = 0; i < extensions.length - 1; i++) {
555                                sb.append(extensions[i]);
556                                sb.append("; *");
557                        }
558                        sb.append(extensions[extensions.length - 1]);
559                        sb.append(")");
560                        return sb.toString();
561                }
