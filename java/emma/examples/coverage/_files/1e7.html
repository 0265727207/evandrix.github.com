494                @Override
495                public void run() {
496                        lastTime = ((double) System.currentTimeMillis()) / 1000 - globalStartTime;
497                        while (!stopped) {
498                                try {
499                                        sleep(1000);
500                                } catch (InterruptedException ie) {
501                                        JOptionPane.showMessageDialog(null, "Unexpected progress timer termination");
502                                }
503                                if (!paused) {
504                                        double time = ((double) System.currentTimeMillis()) / 1000 - globalStartTime;
505                                        elapsedTime += (time - lastTime);
506                                        lastTime = time;
507                                        updateTime();
508                                }
509                        }
510                }
